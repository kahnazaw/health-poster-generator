<h1>مولد الملصقات الصحية</h1>

<textarea id="message" placeholder="اكتب رسالتك هنا..."></textarea>
<button id="generateBtn">عرض الملصق</button>
<button id="downloadBtn">تحميل PDF</button>

<div id="poster" style="display:none;">
  <h1>ملصق صحي</h1>
  <p id="posterText"></p>
</div>

<script>
  const messageInput = document.getElementById("message");
  const poster = document.getElementById("poster");
  const posterText = document.getElementById("posterText");

  // عرض الملصق بالنص الذي كتبه المستخدم
  document.getElementById("generateBtn").addEventListener("click", () => {
    const text = messageInput.value.trim();
    if(text === "") {
      alert("الرجاء كتابة رسالة!");
      return;
    }
    posterText.textContent = text;
    poster.style.display = "block"; // إظهار الملصق
  });

  // تحميل الملصق كـ PDF
  document.getElementById("downloadBtn").addEventListener("click", () => {
    if(poster.style.display === "none") {
      alert("الرجاء عرض الملصق أولاً!");
      return;
    }

    const options = {
      margin: 10,
      filename: 'health-poster.pdf',
      image: { type: 'jpeg', quality: 0.98 },
      html2canvas: { scale: 2, logging: true, useCORS: true },
      jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }
    };

    html2pdf().set(options).from(poster).save();
  });
</script>
